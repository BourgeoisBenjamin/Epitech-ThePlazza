199 void Reception::translateCommand(const std::string &command)
200 try {
201  std::shared_ptr<Order> order = std::make_shared<Order>(command, _cookingMultiplier);
202 
203  std::vector<std::tuple<IPizza::PizzaType, IPizza::PizzaSize, finish_t, send_t>> pizzas = order->getPizzas();
204 
205  std::cout << "[RECEPTION] Order sent to the kitchens" << std::endl;
206  _orders.push_back(order);
207  int a = 0;
208  std::for_each(pizzas.begin(), pizzas.end(), [this, &order, &a](std::tuple<IPizza::PizzaType, IPizza::PizzaSize, finish_t, send_t> &pizza) {
209 
210  // Find a kitchen which can accept the pizza
211  for (int i = 0; static_cast<size_t>(i) < _process.size(); i++) {
212  writeOrderToClient(order, i, pizza);
213  if (clientAcceptOrder(i)) {
214  order->setSend(a, true);
215  a++;
216  return;
217  }
218  }
219  // Fork if no kitchen can take the pizza
220  std::shared_ptr<IIPC> ipc;
221  std::shared_ptr<IProcess> process;
222 
223  if (_type == IIPC::PIPE) {
224  Factory factory(-1, -1, IIPC::PIPE);
225  ipc = factory.createIPC();
226  process = std::make_shared<Process>([this, &process](){
227  process->getIpc()->setRelation(IIPC::CHILD);
228  kitchenProcess(process);
229  }, ipc);
230  } else {
231  process = std::make_shared<Process>([this, &process](){
232  kitchenProcess(process);
233  });
234  }
235  _process.push_back(process);
236  process->runChild();
237  if (_type == IIPC::SOCKET) childConnection();
238  process->getIpc()->setRelation(IIPC::PARENT);
239  writeOrderToClient(order, _process.size() - 1, pizza);
240  if (!clientAcceptOrder(_process.size() - 1))
241  throw ReceptionError("Fatal error : Unable to send the pizza");
242  order->setSend(a, true);
243  a++;
244  });
245 
246 } catch (const ParserError &e) {
247  std::cout << "Invalid command" << std::endl;
248 } catch (const ProcessError &e) {
249  throw e;
250 }
